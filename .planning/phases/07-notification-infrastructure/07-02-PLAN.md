3. **Dedup check (D-13, NOT-08):** Query notification_queue for same dedup_key within last 5 minutes that has already been sent:
```typescript
const fiveMinutesAgo = new Date(Date.now() - 5 * 60 * 1000).toISOString()
const { data: recent } = await supabase
  .from('notification_queue')
  .select('id')
  .eq('dedup_key', record.dedup_key)
  .neq('id', record.id)
  .gte('created_at', fiveMinutesAgo)
  .not('notification_sent_at', 'is', null)
  .limit(1)
  .maybeSingle()
```
If recent exists, update current row status to 'suppressed' and return 200.

4. **Atomic claim (Pitfall 1 prevention):**
```typescript
const { data: claimed } = await supabase
  .from('notification_queue')
  .update({ notification_sent_at: new Date().toISOString(), status: 'processing' })
  .eq('id', record.id)
  .is('notification_sent_at', null)
  .select()
  .single()
```
If claim fails (null or error), return 200 (another invocation claimed it).

5. **Storm cap check (D-14, NOT-08):** Count delivered + processing entries for this event_id in the last hour:
```typescript
const oneHourAgo = new Date(Date.now() - 60 * 60 * 1000).toISOString()
const { count } = await supabase
  .from('notification_queue')
  .select('id', { count: 'exact', head: true })
  .eq('event_id', record.event_id)
  .in('status', ['delivered', 'processing'])
  .gte('created_at', oneHourAgo)
```
If count >= 50, update row to 'suppressed' and return 200.

11. **Update queue row to final status with retry scheduling (D-15):**

Track whether any delivery failed during fan-out (steps 9-10). Then:

```typescript
const anyFailed = /* boolean from delivery tracking */

if (anyFailed) {
  const retryCount = record.retry_count + 1
  if (retryCount >= MAX_RETRIES) {
    // Exhausted all retries — mark permanently failed
    await supabase.from('notification_queue').update({
      status: 'failed',
      retry_count: retryCount,
      next_retry_at: null,
    }).eq('id', record.id)
  } else {
    // Schedule next retry with exponential backoff per D-15
    const backoffMs = RETRY_BACKOFF_MS[record.retry_count] ?? RETRY_BACKOFF_MS[RETRY_BACKOFF_MS.length - 1]
    const nextRetryAt = new Date(Date.now() + backoffMs).toISOString()
    await supabase.from('notification_queue').update({
      status: 'pending',
      retry_count: retryCount,
      next_retry_at: nextRetryAt,
      notification_sent_at: null, // Reset claim so retry can re-claim
    }).eq('id', record.id)
  }
} else {
  // All deliveries succeeded
  await supabase.from('notification_queue').update({ status: 'delivered' }).eq('id', record.id)
}
```
