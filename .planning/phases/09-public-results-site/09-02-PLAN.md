interface Props {
  games: PublicGame[]

  - Active style: `bg-[#0B3D91] text-white rounded-md px-3 py-2`

  - Font: `font-cond text-[10px] font-bold tracking-[.1em] uppercase`

- Day navigation row (per D-05): `<div className="flex items-center gap-1 overflow-x-auto pb-1 mb-4">` (add CSS class `no-scrollbar` if needed)
  - Each day derived from `eventDates` sorted by `day_number`
  - Link: `href={/e/${slug}?tab=schedule&view=${view}&day=${date.day_number}&div=${divFilter}}`
  - Label: `Day ${date.day_number}` (or `date.label` if available)
  - Active/inactive same pill style as toggle row

- Conditionally render `<ByTeamView>`, `<ByFieldView>`, or `<ByTimeView>` based on `view` prop
