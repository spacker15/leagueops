    ```typescript
    import { NextResponse } from 'next/server'
    import { engineRatelimit } from '@/lib/ratelimit'

    export async function POST(request: Request) {
      // Rate limit by IP (SEC-08)
      const ip =
        request.headers.get('x-forwarded-for')?.split(',')[0]?.trim() ??
        '127.0.0.1'
      const { success, limit, remaining, reset, pending } =
        await engineRatelimit.limit(ip)
      void pending

      if (!success) {
        return NextResponse.json(
          { error: 'Too many requests' },
          {
            status: 429,
            headers: {
              'X-RateLimit-Limit': String(limit),
              'X-RateLimit-Remaining': String(remaining),
              'X-RateLimit-Reset': String(reset),
            },
          }
        )
      }

      // ... rest of existing handler (auth guard, business logic) ...
    }
    ```

    Rules:
    - Use `engineRatelimit` (NOT `publicRatelimit`) for engine routes — 10 req/60s fixed window
    - Extract IP from `x-forwarded-for` header, taking ONLY the first IP: `split(',')[0]?.trim()`
    - Fall back to `'127.0.0.1'` if header is absent
    - Include `void pending` to flush analytics without blocking (safe no-op on non-Vercel)
    - 429 response body is `{ error: 'Too many requests' }`
    - 429 response MUST include three headers: `X-RateLimit-Limit`, `X-RateLimit-Remaining`, `X-RateLimit-Reset`
    - All header values MUST be converted to String: `String(limit)`, `String(remaining)`, `String(reset)`
    - Rate limit check goes BEFORE auth guard (cheaper to reject early)
    - Preserve all existing handler logic after the rate limit block
    - Do NOT create a new Ratelimit instance inside the handler — import the module-level one from lib/ratelimit.ts
